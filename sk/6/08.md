          uint randNonce = 0;
          uint attackVictoryProbability = 70;

          function randMod(uint _modulus) internal returns(uint) {
            randNonce++;
            return uint(keccak256(now, msg.sender, randNonce)) % _modulus;

          function attack(uint _zombieId, uint _targetId) external onlyOwnerOf(_zombieId) {
            Zombie storage myZombie = zombies[_zombieId];

            uint rand = randMod(100);

              myZombie.level++;
              enemyZombie.lossCount++;
              feedAndMultiply(_zombieId, enemyZombie.dna, "zombie");
            } else {
              myZombie.lossCount++;
              enemyZombie.winCount++;

## Zvyšujeme level!
## Level Up!

Spomeň si na `ZombieHelper`. Pridali sme tam `payable` funkciu, aby si užívateľ mohol zvýšiť level:
Recall in `ZombieHelper`, we added a payable function where the user can level up:

```
function levelUp(uint _zombieId) external payable {
  require(msg.value == levelUpFee);
  zombies[_zombieId].level++;
}
```

Spôsob ako poslať Ether spolu s volaním funkcie kontraktu je jednoduchý, no je v tom jeden háčik. Namiesto toho aby sme špecifikovali koľko chceme poslať Etheru, musíme určiť, koľko chceme poslať `wei`.

## Co je to Wei?

`wei` je najmenšia, ďalej nedeliteľná jednotka Etheru. V propočte sa 10^18 `wei` rovná 1 `ether`.

To je veľa núl na rátanie - našťastie Web3.js pre nás má nástroj na konverziu medzi tými jednotkami. 

```
// Toto skonvertuje 1 ETH na Wei
web3js.utils.toWei("1", "ether");
```

V našom kontrakte sme nastavovali `levelUpFee = 0.001 ether`, takže keď voláme funkciu `levelUp`, môžeme zaručiť že užívateľ odošle `0.001` Etheru takýmto kódom:

```
CryptoZombies.methods.levelUp(zombieId)
.send({ from: userAccount, value: web3js.utils.toWei("0.001", "ether") })
```

## Vyskúšaj si to sám

Poďme pridať funkciu `levelUp` pod `feedOnKitty`. Kód bude veľmi podobný `feedOnKitty`, no:

1. Funkcia bude príjmať len 1 parameter, `zombieId`

2. Pred odoslaním transakcie by `txStatus` mal obstahovať text `"Leveling up your zombie..."`

3. Počas volania `levelUp` funkcie kontrakte by sa malo k transakcii pripojiť `"0.001"` ETH skonvertovaného pomocou `toWei`, tak ako na príklade vyššie

4. V prípade úspešnej transakcie by sa mal zobraziť text `"Power overwhelming! Zombie successfully leveled up"`

5. **Nemusíme** sa dotazovať náš kontrakt na dáta z `getZombiesByOwner`. V tomto prípade totiž vieme, že jediná vec ktorá sa po transakcií zmení, je level našeho zombie.
