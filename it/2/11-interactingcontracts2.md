            uint dnaModulus = 10 ** dnaDigits;

                string name;

                uint id = zombies.push(Zombie(_name, _dna)) - 1;
                zombieToOwner[id] = msg.sender;
                ownerZombieCount[msg.sender]++;

Continuando il nostro esempio precedente con `NumberInterface`, una volta definita l'interfaccia come:

```
contract NumberInterface {
  function getNum(address _myAddress) public view returns (uint);
}
```

Possiamo usarlo in un contratto come segue:

```
contract MyContract {
  address NumberInterfaceAddress = 0xab38... 
  // ^ L'indirizzo del contratto FavoriteNumber su Ethereum
  NumberInterface numberContract = NumberInterface(NumberInterfaceAddress);
  // Ora `numberContract` punta all'altro contratto

  function someFunction() public {
    // Ora possiamo chiamare `getNum` da quel contratto:
    uint num = numberContract.getNum(msg.sender);
    // ...e fai qualcosa con `num` qui
  }
}
```

In questo modo, il tuo contratto può interagire con qualsiasi altro contratto sulla blockchain di Ethereum, purché espongano tali funzioni come `public` o `external`.
